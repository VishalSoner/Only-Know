class C
{
	public static void main(String args[])
	{
		int x[] = {1, 2, 3, 5, 4, 9, 8, 7};
		
		find_Miss_Element(x, x.length);

	}
	
	
	static void find_Miss_Element(int x[], int n)
	{
		if(n == 1)
			return;
		
		int temp[] = new int[n+2];
		
		for(int i = 0; i < n; i++)
			temp[ x[i] ] = 1;

		for(int i = 1; i < temp.length-1; i++)
		{
			if(temp[i] == 0)
			{
				System.out.println( i );
				break;
			}
		}
	}
}
